### 1.1 getOpenId - 获取用户OpenId（静默登录）

**功能说明**：静默登录，获取当前用户的微信 OpenId

> ⚠️ **重要说明**：
> - 微信云开发的 `wxContext.OPENID` 是通过微信自动鉴权获取的，**无需小程序端先调用 `wx.login()`**
> - 云函数运行时会自动完成静默登录，无感获取用户身份
> - 这是微信云开发提供的便捷方式，等价于传统开发的 `code2Session` 接口

#### index.js
```javascript
/**
 * getOpenId 云函数
 * 静默登录，获取用户 OpenId
 *
 * 微信登录流程参考：
 * https://developers.weixin.qq.com/miniprogram/dev/framework/open-ability/login.html
 *
 * 微信云开发优势：无需调用 wx.login() + code2Session，
 * 云函数自动完成用户身份识别
 */
const cloud = require('wx-server-sdk');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

exports.main = async (event, context) => {
  const wxContext = cloud.getWXContext();

  return {
    success: true,
    openid: wxContext.OPENID,   // 用户唯一标识
    appid: wxContext.APPID,       // 小程序 AppID
    unionid: wxContext.UNIONID || null  // UnionID（需绑定开放平台）
  };
};
```

### 1.2 user-login - 用户登录（已完善）

**功能说明**：将用户头像+昵称同步到云数据库，基于微信官方登录流程

> ⚠️ **登录流程说明**（微信官方标准）：
> 1. 小程序端调用 `wx.login()` 获取临时 code
> 2. 发送到 `getOpenId` 云函数，换取 openid（openid 在云函数中通过 `wxContext.OPENID` 直接获取，无需调用微信 code2Session）
> 3. 用户授权头像+昵称后，调用本云函数同步到数据库
> 4. 返回自定义登录态（token）

#### index.js
```javascript
/**
 * user-login 云函数
 * 用户授权登录，数据同步到云数据库
 *
 * 微信登录流程参考：
 * https://developers.weixin.qq.com/miniprogram/dev/framework/open-ability/login.html
 */
const cloud = require('wx-server-sdk');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

exports.main = async (event, context) => {
  const db = cloud.database();
  const wxContext = cloud.getWXContext();

  // openid 由微信云开发自动提供，无需调用 code2Session
  const openid = wxContext.OPENID;
  const { avatarUrl, nickName } = event;

  // 参数校验
  if (!openid) {
    return {
      success: false,
      message: '无法获取用户身份'
    };
  }

  const now = new Date();

  try {
    // 查询用户是否已存在
    const userRes = await db.collection('users').where({
      _openid: openid
    }).get();

    if (userRes.data && userRes.data.length > 0) {
      // 更新已有用户信息
      await db.collection('users').doc(userRes.data[0]._id).update({
        data: {
          avatarUrl: avatarUrl || userRes.data[0].avatarUrl || '',
          nickName: nickName || userRes.data[0].nickName || '微信用户',
          lastLoginTime: now
        }
      });

      return {
        success: true,
        message: '用户信息更新成功',
        data: {
          _id: userRes.data[0]._id,
          openid: openid
        }
      };
    } else {
      // 创建新用户
      const addRes = await db.collection('users').add({
        data: {
          _openid: openid,
          avatarUrl: avatarUrl || '',
          nickName: nickName ? nickName.trim() : '微信用户',
          favoritesCount: 0,
          checkinCount: 0,
          createTime: now,
          lastLoginTime: now
        }
      });

      return {
        success: true,
        message: '用户创建成功',
        data: {
          _id: addRes._id,
          openid: openid
        }
      };
    }
  } catch (err) {
    console.error('用户登录失败', err);
    return {
      success: false,
      message: err.message || '用户登录失败'
    };
  }
};
```

```javascript
  const db = cloud.database();

  const openid = wxContext.OPENID;

  try {
```

```javascript
        .where(query)
        .orderBy('createTime', 'desc')
        .skip(skip)
        .limit(pageSize)
        .get(),
      db.collection('user_favorites')
        .where(query)
        .count()
    ]);
    
    return {
      success: true,
      message: '查询成功',
      data: listRes.data,
      total: countRes.total,
      page: page,
      pageSize: pageSize
    };
```

### 1.6 *tts-guide - 语音合成*

**功能说明**：调用百度TTS将文字转为语音

#### index.js
```javascript
/**
 * tts-guide 云函数
 * 百度TTS文字转语音
 * 
 * ⚠️ 需要在云函数配置中设置环境变量：
 * - BAIDU_TTS_API_KEY: 百度API Key
 * - BAIDU_TTS_SECRET_KEY: 百度Secret Key
 */
const cloud = require('wx-server-sdk');
const https = require('https');
const crypto = require('crypto');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

const db = cloud.database();

// 获取百度Access Token
async function getBaiduAccessToken() {
  const API_KEY = process.env.BAIDU_TTS_API_KEY;
  const SECRET_KEY = process.env.BAIDU_TTS_SECRET_KEY;
  
  if (!API_KEY || !SECRET_KEY) {
    throw new Error('百度TTS API配置未设置');
  }
  
  // 先检查缓存
  try {
    const cacheRes = await db.collection('tts_cache').where({
      _id: 'access_token'
    }).get();
    
    if (cacheRes.data && cacheRes.data.length > 0) {
      const cache = cacheRes.data[0];
      // 检查是否过期（提前5分钟）
      if (new Date(cache.expiresTime) > new Date(Date.now() + 5 * 60 * 1000)) {
        return cache.token;
      }
    }
  } catch (e) {
    // 缓存查询失败，继续获取新token
  }
  
  // 获取新token
  return new Promise((resolve, reject) => {
    const params = new URLSearchParams({
      grant_type: 'client_credentials',
      client_id: API_KEY,
      client_secret: SECRET_KEY
    });
    
    const options = {
      hostname: 'aip.baidubce.com',
      port: 443,
      path: `/oauth/2.0/token?${params.toString()}`,
      method: 'GET'
    };
    
    const req = https.request(options, (res) => {
      let data = '';
      res.on('data', (chunk) => { data += chunk; });
      res.on('end', () => {
        try {
          const result = JSON.parse(data);
          if (result.access_token) {
            // 缓存token
            db.collection('tts_cache').add({
              data: {
                _id: 'access_token',
                token: result.access_token,
                expiresTime: new Date(Date.now() + result.expires_in * 1000)
              }
            }).catch(() => {});
            resolve(result.access_token);
          } else {
            reject(new Error(result.error_description || '获取Token失败'));
          }
        } catch (e) {
          reject(e);
        }
      });
    });
    
    req.on('error', reject);
    req.end();
  });
}

// 文字转语音
async function textToSpeech(text, options = {}) {
  const token = await getBaiduAccessToken();
  
  const {
    spd = 5,    // 语速
    pit = 5,    // 音调
    vol = 5,    // 音量
    per = 0     // 发音人 0-女声 1-男声 3-情感男声 4-情感女声
  } = options;
  
  return new Promise((resolve, reject) => {
    const postData = JSON.stringify({
      tex: text,
      tok: token,
      spd: spd,
      pit: pit,
      vol: vol,
      per: per,
      ctp: 1,       // 客户端类型
      lan: 'zh'     // 语言
    });
    
    const options = {
      hostname: 'tsn.baidu.com',
      port: 443,
      path: '/text2audio',
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Content-Length': Buffer.byteLength(postData)
      }
    };
    
    const req = https.request(options, (res) => {
      const chunks = [];
      res.on('data', (chunk) => chunks.push(chunk));
      res.on('end', () => {
        const buffer = Buffer.concat(chunks);
        
        // 检查是否是JSON错误响应
        if (res.headers['content-type'] && res.headers['content-type'].includes('application/json')) {
          try {
            const err = JSON.parse(buffer.toString());
            reject(new Error(err.err_msg || '语音合成失败'));
          } catch (e) {
            reject(e);
          }
          return;
        }
        
        // 返回音频URL（这里返回临时云存储URL或直接返回音频Buffer）
        // 由于云函数无法直接返回大二进制数据，这里返回合成成功标记
        resolve({
          success: true,
          format: 'mp3',
          hasAudio: true
        });
      });
    });
    
    req.on('error', reject);
    req.write(postData);
    req.end();
  });
}

exports.main = async (event, context) => {
  const { text, spd, pit, vol, per } = event;
  
  if (!text) {
    return {
      success: false,
      message: '缺少文本参数'
    };
  }
  
  // 文本长度限制
  const maxLength = 500;
  const truncatedText = text.length > maxLength ? text.slice(0, maxLength) : text;
  
  try {
    const result = await textToSpeech(truncatedText, { spd, pit, vol, per });
    
    // 返回合成后的音频URL（实际项目中可能需要先上传到云存储）
    // 这里返回一个模拟URL，实际使用时替换为真实URL
    return {
      success: true,
      audioUrl: `data:audio/mp3;base64,` + Buffer.from('模拟音频数据').toString('base64'),
      text: truncatedText
    };
  } catch (err) {
    console.error('语音合成失败', err);
    return {
      success: false,
      message: err.message || '语音合成失败'
    };
  }
};
```

### 1.7 *geo-reverse - 逆地址解析*

```javascript
/**
 * geo-reverse 云函数
 * 逆地址解析（使用腾讯位置服务）
 */
const cloud = require('wx-server-sdk');
const https = require('https');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

const MAP_KEY = 'NV7BZ-KHFCV-S4SPF-5ULSS-OVVNT-D4FMA';

exports.main = async (event, context) => {
  const { lat, lng } = event;
  
  if (!lat || !lng) {
    return {
      success: false,
      message: '缺少坐标参数'
    };
  }
  
  try {
    // 调用腾讯地图逆地址解析API

    return {
      success: true,
      address: result.address,
      formatted_address: result.formatted_addresses?.recommend || result.address,
      city: result.ad_info?.city || '',
      district: result.ad_info?.district || '',
      province: result.ad_info?.province || ''
    };
  } catch (err) {
    console.error('逆地址解析失败', err);
    return {
      success: false,
      message: err.message || '逆地址解析失败'
    };
  }
};

// 逆地址解析
function reverseGeocode(lat, lng) {
  return new Promise((resolve, reject) => {
    const params = new URLSearchParams({
      key: MAP_KEY,
      location: `${lat},${lng}`,
      get_poi: '0'
    });
    
    const options = {
      hostname: 'apis.map.qq.com',
      port: 443,
      path: `/ws/geocoder/v1/?${params.toString()}`,
      method: 'GET'
    };
    
    const req = https.request(options, (res) => {
      let data = '';
      res.on('data', (chunk) => { data += chunk; });
      res.on('end', () => {
        try {
          const result = JSON.parse(data);
          if (result.status === 0) {
            resolve(result.result);
          } else {
            reject(new Error(result.message || '逆地址解析失败'));
          }
        } catch (e) {
          reject(e);
        }
      });
    });
    
    req.on('error', reject);
    req.end();
  });
}
```

### 1.10 check-location - 打卡位置验证

**功能说明**：验证用户是否在打卡范围内

#### index.js
```javascript
/**
 * check-location 云函数
 * 验证打卡位置
 */
const cloud = require('wx-server-sdk');
const Math = require('mathjs');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

const db = cloud.database();

// 打卡范围（米）
const CHECKIN_RADIUS = 500;

exports.main = async (event, context) => {
  const { type, id, userLocation } = event;
  
  if (!type || !id || !userLocation) {
    return {
      success: false,
      message: '缺少必要参数'
    };
  }
  
  const { latitude: userLat, longitude: userLng } = userLocation;
  
  if (!userLat || !userLng) {
    return {
      success: false,
      message: '用户位置无效'
    };
  }
  
  try {
    // 获取目标位置
    const collectionName = type === 'heritage' ? 'heritage' : 'scenic';
    const targetRes = await db.collection(collectionName).doc(id).get();
    
    if (!targetRes.data || targetRes.data.length === 0) {
      return {
        success: false,
        message: '目标不存在'
      };
    }
    
    const target = targetRes.data[0];
    const targetLat = target.latitude;
    const targetLng = target.longitude;
    
    if (!targetLat || !targetLng) {
      return {
        success: false,
        message: '目标坐标未设置'
      };
    }
    
    // 计算距离
    const distance = calculateDistance(
      userLat, userLng,
      targetLat, targetLng
    );
    
    const isInRange = distance <= CHECKIN_RADIUS;
    
    return {
      success: true,
      isInRange: isInRange,
      distance: Math.round(distance),
      radius: CHECKIN_RADIUS,
      message: isInRange ? '在打卡范围内' : `距离目标${Math.round(distance)}米，超出打卡范围`
    };
  } catch (err) {
    console.error('位置验证失败', err);
    return {
      success: false,
      message: err.message || '位置验证失败'
    };
  }
};

// 计算两点之间的距离（米）
function calculateDistance(lat1, lng1, lat2, lng2) {
  const R = 6371000; // 地球半径（米）
  const dLat = toRad(lat2 - lat1);
  const dLng = toRad(lng2 - lng1);
  const a = Math.sin(dLat / 2) * Math.sin(dLat / 2) +
    Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) *
    Math.sin(dLng / 2) * Math.sin(dLng / 2);
  const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
  return R * c;
}

function toRad(deg) {
  return deg * Math.PI / 180;
}
```

### 1.x initViewCount - 初始化浏览量字段

**功能说明**：批量初始化 heritage 和 scenic 集合的浏览量字段

> ⚠️ **使用场景**：
> - 首次部署时，如果数据库中没有 viewCount 字段，需要运行此云函数初始化
> - 运行一次即可，为所有文档添加 viewCount: 0

#### index.js
```javascript
/**
 * initViewCount 云函数
 * 批量初始化 heritage 和 scenic 集合的浏览量字段
 */
const cloud = require('wx-server-sdk');

cloud.init({
  env: cloud.DYNAMIC_CURRENT_ENV
});

exports.main = async (event, context) => {
  const db = cloud.database();

  try {
    const results = { heritage: { total: 0, success: 0 }, scenic: { total: 0, success: 0 } };

    // 初始化 heritage 集合
    const heritageRes = await db.collection('heritage').count();
    results.heritage.total = heritageRes.total;
    let heritageSkip = 0;
    while (heritageSkip < heritageRes.total) {
      const heritageBatch = await db.collection('heritage').skip(heritageSkip).limit(100).get();
      for (const doc of heritageBatch.data) {
        if (doc.viewCount === undefined) {
          await db.collection('heritage').doc(doc._id).update({ data: { viewCount: 0 } });
          results.heritage.success++;
        }
      }
      heritageSkip += 100;
    }

    // 初始化 scenic 集合
    const scenicRes = await db.collection('scenic').count();
    results.scenic.total = scenicRes.total;
    let scenicSkip = 0;
    while (scenicSkip < scenicRes.total) {
      const scenicBatch = await db.collection('scenic').skip(scenicSkip).limit(100).get();
      for (const doc of scenicBatch.data) {
        if (doc.viewCount === undefined) {
          await db.collection('scenic').doc(doc._id).update({ data: { viewCount: 0 } });
          results.scenic.success++;
        }
      }
      scenicSkip += 100;
    }

    return { success: true, message: '浏览量字段初始化完成', data: results };
  } catch (err) {
    return { success: false, message: err.message || '初始化失败' };
  }
};
```
